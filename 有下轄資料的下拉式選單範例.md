**有下轄資料的下拉式選單範例**
___
```html
<!-- 單位名稱 -->
<div class="input-group">
  <div class="form-group me-3 mb-4 required">
    <label for="companyNameSelect" class="form-label">單位名稱</label>
    <div class="d-flex flex-row align-items-center">
      <validation-provider
        v-slot="{ classes, errors }"
        :rules="{ required: true }"
        name="單位名稱"
      >
        <select
          id="companyNameSelect"
          v-model="formInput.companyName"
          :class="classes"
          class="form-select"
        >
          <option
            v-for="option in options.companyNameArray"
            :key="option.id"
            :value="option.id"
          >
            {{ option.title }}
          </option>
        </select>
        <span class="text-danger position-absolute">{{
          errors[0]
        }}</span>
      </validation-provider>
      <input
        class="form-check-input ms-5"
        type="checkbox"
        id="isSummaryJurisdictionCheck"
        v-model="formInput.isSummaryJurisdiction"
      />
      <label
        class="form-check-label ms-2"
        for="isSummaryJurisdictionCheck"
      >
        彙總下轄單位資料
      </label>
    </div>
  </div>
</div>
<!-- 下轄資料 -->
<div class="input-group" v-if="!formInput.isSummaryJurisdiction">
  <div class="form-group me-3 mb-4 required">
    <label for="jurisdictionStartSelect" class="form-label"
      >轄屬單位</label
    >
    <div class="d-flex flex-row align-items-center">``
      <validation-provider
        v-slot="{ classes, errors }"
        :rules="{ required: true }"
        name="轄屬單位(起)"
      >
        <select
          id="jurisdictionStartSelect"
          v-model="formInput.jurisdictionStart"
          :class="classes"
          class="form-select"
        >
          <option value="">選擇值</option>
          <option
            v-for="option in options.jurisdictionStartArray"
            :key="option.id"
            :value="option.id"
          >
            {{ option.title }}
          </option>
        </select>
        <span class="text-danger position-absolute">{{
          errors[0]
        }}</span>
      </validation-provider>
      <nobr><span class="mx-2">~</span></nobr>
      <validation-provider
        v-slot="{ classes, errors }"
        :rules="{ required: true }"
        name="轄屬單位(迄)"
      >
        <select
          id="jurisdictionEndSelect"
          v-model="formInput.jurisdictionEnd"
          :class="classes"
          class="form-select"
        >
          <option value="">選擇值</option>
          <option
            v-for="option in options.jurisdictionEndArray"
            :key="option.id"
            :value="option.id"
          >
            {{ option.title }}
          </option>
        </select>
        <span class="text-danger position-absolute">{{
          errors[0]
        }}</span>
      </validation-provider>
    </div>
  </div>
</div>
```
___
script部分
```html
<script>
export default {
  //資料
  data() {
    return {
      //資料(該畫面輸入資料或是要顯示的資料)
      formInput: {
        //單位名稱
        companyName: '0',
        //是否彙總下轄資料
        isSummaryJurisdiction: true,
        //下轄資料(起)
        jurisdictionStart: '',
        //下轄資料(迄)
        jurisdictionEnd: '',
      },

      //選項(下拉式選單用)
      options: {
        //單位名稱列表
        companyNameArray: [
          { id: '0', title: '區處別AAA' },
          { id: '1', title: '區處別BBB' },
          { id: '2', title: '區處別CCC' },
          { id: '3', title: '區處別DDD' },
          { id: '4', title: '區處別EEE' },
        ],
        //下轄資料(起)列表
        jurisdictionStartArray: [
          { id: '0', title: '下轄資料(起)AAA' },
          { id: '1', title: '下轄資料(起)BBB' },
          { id: '2', title: '下轄資料(起)CCC' },
          { id: '3', title: '下轄資料(起)DDD' },
          { id: '4', title: '下轄資料(起)EEE' },
        ],
        //下轄資料(迄)列表
        jurisdictionEndArray: [
          { id: '0', title: '下轄資料(迄)AAA' },
          { id: '1', title: '下轄資料(迄)BBB' },
          { id: '2', title: '下轄資料(迄)CCC' },
          { id: '3', title: '下轄資料(迄)DDD' },
          { id: '4', title: '下轄資料(迄)EEE' },
        ],
      },
    };
  },
};
</script>

```
