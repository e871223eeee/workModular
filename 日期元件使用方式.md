**日期元件使用方式**

先在<script>區域引入該元件並定義
```C#
<script>
//引入時間選擇器模塊
import DatePickerEx from "@/components/Util/DatePickerEx.vue";

export default {
  //名稱
  name: "XXXXX",
  components: {
    //定義時間選擇器模塊
    DatePickerEx,
  },
};
</script>
```

___
- 如果要連動function -> 加入 @input=""
- CustomValue_1 : 名子有對到就好
- CustomValue_2 : 綁定的變數名稱
- CustomValue_3 : 輸出格式
  - yearFormatter : YYY
  - yearMonthFormatter : YYY/MM
  - defaultFormatter : YYY/MM/DD
- CustomValue_4 : 選單格式
  - year : 選年
  - month : 選年月
  - 直接不填該欄位: 選年月日

___
沒有輸入驗證版本
```html
<div class="input-group">
  <div class="form-group me-3 mb-4">
    <label for="CustomValue_1 DatePicker" class="form-label">CustomValue_選擇器標題</label>
    <DatePickerEx
      v-model="formInput.CustomValue_2"
      :input-attr="{ id: 'CustomValue_1 DatePicker' }"
      :input-class="{ 'form-control': true }"
      :formatter="CustomValue_3"
      placeholder="CustomValue_未輸入時顯示"
      output="number"
      type="CustomValue_4"
      class="d-block"
    />
  </div>
</div>
```
___
輸入驗證版本
```html
<div class="input-group">
  <div class="form-group me-3 mb-4 required">
    <label for="CustomValue_1 DatePicker" class="form-label">CustomValue_選擇器標題</label>
    <validation-provider
      v-slot="{ classes, errors }"
      :rules="{ required: true }"
      name="CustomValue_選擇器標題"
    >
      <DatePickerEx
        v-model="formInput.CustomValue_2"
        :input-attr="{ id: 'CustomValue_1 DatePicker' }"
        :input-class="{ ...classes, 'form-control': true }"
        :formatter="CustomValue_3"
        placeholder="CustomValue_未輸入時顯示"
        type="CustomValue_4"
        output="number"
        class="d-block"
      />
      <span class="text-danger position-absolute">{{ errors[0] }}</span>
    </validation-provider>
  </div>
</div>
```

___
起訖範例
```html
<!-- 年度 -->
<div class="input-group">
  <div class="form-group me-3 mb-4">
    <label for="yearStartDatePicker" class="form-label">年度</label>
    <div class="d-flex flex-row align-items-center">
      <DatePickerEx
        v-model="formInput.yearStart"
        :input-attr="{ id: 'yearStartDatePicker' }"
        :input-class="{ 'form-control': true }"
        :formatter="yearFormatter"
        placeholder="年度(起)"
        type="year"
        output="number"
        class="d-block"
        @input="yearStartDatePickerChange"
      />
      <nobr><span class="mx-2">~</span></nobr>
      <DatePickerEx
        v-model="formInput.yearEnd"
        :input-attr="{ id: 'yearEndDatePicker' }"
        :input-class="{ 'form-control': true }"
        :formatter="yearFormatter"
        placeholder="年度(訖)"
        type="year"
        output="number"
        class="d-block"
        @input="yearEndDatePickerChange"
      />
    </div>
  </div>
</div>
```
script範例
```html
<script>
//引入時間選擇器模塊
import DatePickerEx from '@/components/Util/DatePickerEx.vue';
export default {
  components: {
    //定義時間選擇器模塊
    DatePickerEx,
  },
  //資料
  data() {
    return {
      //資料(該畫面輸入資料或是要顯示的資料)
      formInput: {
        //單位名稱
        companyName: '0',
        //是否彙總下轄資料
        isSummaryJurisdiction: true,
        //下轄資料(起)
        jurisdictionStart: '',
        //下轄資料(迄)
        jurisdictionEnd: '',
        //年度(起)
        yearStart: null,
        //年度(迄)
        yearEnd: null,
      },
    };
  },

  //畫面初始化
  mounted() {},

  //方法
  methods: {
    //選到年度(起)時，如果年度(迄)比它小，那值就跟年度(起)一樣
    yearStartDatePickerChange() {
      if (this.formInput.yearStart > this.formInput.yearEnd)   {
        this.formInput.yearEnd = this.formInput.yearStart;
      }
    },
    //選到年度(迄)時，如果年度(起)比它大，那值就跟年度(迄)一樣
    yearEndDatePickerChange() {
      if (this.formInput.yearStart > this.formInput.yearEnd)   {
        this.formInput.yearStart = this.formInput.yearEnd;
      }
    },
  },
};
</script>
```
